// check if number is prime
function is_prime(n) {
    if (n <= 1) {
        return false;
    }
    let i = 2;
    while (i < n) {
        // do something
        if (n % i == 0) {
            return false;
        }
        i = i + 1;
    }
    return true;
}

print(is_prime(7));
print(is_prime(10));
