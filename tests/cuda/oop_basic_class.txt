// Test: OOP with CUDA
// Tests class definition and instantiation

print "=== OOP: Basic Class on CUDA ===";

use_cuda {
    class Point {
        public x;
        public y;
        
        function constructor(x_val, y_val) {
            this.x = x_val;
            this.y = y_val;
        }
        
        public function distance() {
            return (this.x * this.x) + (this.y * this.y);
        }
        
        public function display() {
            print "Point:";
            print this.x;
            print this.y;
        }
    }
    
    let p1 = new Point(5, 10);
    p1.display();
    
    let p2 = new Point(15, 20);
    p2.display();
}

print "Basic class test complete!";
