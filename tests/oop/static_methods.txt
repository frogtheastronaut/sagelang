// Test static methods
// Static methods are called on the class, not instances

class MathUtils {
    static function add(a, b) {
        return a + b;
    }
    
    static function multiply(a, b) {
        return a * b;
    }
    
    static function factorial(n) {
        if n <= 1 {
            return 1;
        }
        return n * MathUtils.factorial(n - 1);
    }
}

class Counter {
    function constructor(name) {
        this.name = name;
        this.id = Counter.nextId(0);
    }
    
    static function nextId(dummy) {
        return 1;
    }
    
    public function display(dummy) {
        print this.name + " (ID: " + this.id + ")";
    }
}

// Call static methods on class
print "5 + 3 = " + MathUtils.add(5, 3);
print "5 * 3 = " + MathUtils.multiply(5, 3);
print "5! = " + MathUtils.factorial(5);

print "---";

// Static method usage in constructors
let c1 = Counter("First");
let c2 = Counter("Second");
let c3 = Counter("Third");

c1.display(0);
c2.display(0);
c3.display(0);
